# ComboBox

An accessible ComboBox Component with

1. single
2. multi selectable

with typeahead & autocomplete support. (Accessible in future)

## Import

```js
import { ComboBox } from "@chakra-components/core";
import axios from 'axios'


```
## Async fetching

```jsx
function ComboBoxExample() {
  const fruits = [
    "Acai",
    "Apples",
    "Apricots",
    "Avocado",
    "Ackee",
    "Bananas",
  ];

  const asyncFetchOptions = (value, successCb, errorCb) => {
    axios.get(`https://restcountries.eu/rest/v2/name/${value}`)
      .then(response => {
        successCb(response.data.map(eachpiece => ({
          label: `${eachpiece.name}`,
          value: `${eachpiece.name}`
        })))
      })
      .catch((e) => {
        errorCb('Custom error message')
      })
  }

  const [fruit, setFruit] = React.useState(null);
  return (
    <>
      <ComboBox
        async
        value={fruit}
        onChange={setFruit}
        loadOptions={asyncFetchOptions}
        defaultOptions={[
          // Show default options
          { label: 'default-option-1', value: 'default-option-1' }, 
          { label: 'default-option-2', value: 'default-option-2' }, 
          { label: 'default-option-3', value: 'default-option-3' }
        ]}
      >
        <ComboBoxRightElement
          children={<Icon name="chevron-down" fontSize="1.5rem" />}
        />
        <ComboBoxClearElement />
        <ComboBoxInput placeholder="Select a fruit..." />
        <ComboBoxSelectedGhost />
        <ComboBoxPopper>
          <ComboBoxList>
            <ComboBoxOption />
          </ComboBoxList>
        </ComboBoxPopper>
      </ComboBox>
    </>
  );
}
```
