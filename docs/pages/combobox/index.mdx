# ComboBox

An accessible ComboBox Component with

1. single
2. multi selectable

with typeahead & autocomplete support. (Accessible in future)

## Import

```js
import { ComboBox } from "@chakra-components/core";
```

## Usage

Here's a basic usage of the Select component.

```jsx
<ComboBox>
  <ComboBoxLeftElement color="gray.300" fontSize="1.2em" children="$" />
  <ComboBoxRightElement
    children={<Icon name="chevron-down" fontSize="1.5rem" />}
  />
  <ComboBoxClearElement />
  <ComboBoxInput as={Select} placeholder="Enter amount">
    <option value="option1">Option 1</option>
    <option value="option2">Option 2</option>
    <option value="option3">Option 3</option>
  </ComboBoxInput>
</ComboBox>
```

## Async fetching

```jsx
function ComboBoxExample() {
  const fruits = [
    "Acai",
    "Apples",
    "Apricots",
    "Avocado",
    "Ackee",
    "Bananas",
  ];

  const _axios = ({ speed }) => new Promise(resolve => {
    setTimeout(() => {
      resolve([
        'dynamic-option-1', 
        'dynamic-option-2', 
        'dynamic-option-3', 
        'dynamic-option-4', 
        'dynamic-option-5',
        'dynamic-option-6', 
        'dynamic-option-7', 
        'dynamic-option-8', 
        'dynamic-option-9', 
        'dynamic-option-10',
        'dynamic-option-11', 
        'dynamic-option-12', 
        'dynamic-option-13', 
        'dynamic-option-14', 
        'dynamic-option-15',
        'dynamic-option-16', 
        'dynamic-option-17', 
        'dynamic-option-18', 
        'dynamic-option-19', 
        'dynamic-option-20',
      ])
    }, speed === 'slow' ? 4000 : 1000)
    
  })

  const asyncFetchOptions = (value, callback) => {
    _axios(({
      speed: 'fast' // 'fast' 
    })).then((data) => {
      callback(data.map((d, i) => ({
        value: d,
        label: d,
      })))
    })
  }

  const [fruit, setFruit] = React.useState(null);
  return (
    <>
      <ComboBox
        async
        value={fruit}
        onChange={setFruit}
        loadOptions={asyncFetchOptions}
        defaultOptions={[{ label: 'default-option-1', value: 'default-option-1' }, { label: 'default-option-2', value: 'default-option-2' }, { label: 'default-option-3', value: 'default-option-3' }]}
      >
        <ComboBoxRightElement
          children={<Icon name="chevron-down" fontSize="1.5rem" />}
        />
        <ComboBoxClearElement />
        <ComboBoxInput placeholder="Select a fruit..." />
        <ComboBoxSelectedGhost />
        <ComboBoxPopper>
          <ComboBoxList>
            <ComboBoxOption />
          </ComboBoxList>
        </ComboBoxPopper>
      </ComboBox>
    </>
  );
}
```
