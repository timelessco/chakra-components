function MultiSelectExample() {
  const [defaultOptions, setDefaultOptions] = useState([]);

  const loadOptions = (inputValue, successCb, errorCb) => {
    axios
      .get(`https://restcountries.eu/rest/v2/name/${inputValue}`)
      .then(response => {
        successCb(
          response.data.map(ep => ({
            label: `${ep.name}`,
            value: `${ep.name}`,
            code: `${ep.alpha2Code}`,
            timeZone: `${ep.timezones[0]}`,
          })),
        );
      })
      .catch(e => {
        errorCb("Remote fetch failed");
      });
  };

  useEffect(() => {
    axios.get(`https://restcountries.eu/rest/v2/all`).then(response => {
      setDefaultOptions(
        response.data.map(ep => ({
          label: `${ep.name}`,
          value: `${ep.name}`,
          code: `${ep.alpha2Code}`,
          timeZone: `${ep.timezones[0]}`,
        })),
      );
    });
  }, []);

  const [country, setCountry] = useState(null);

  return (
    <MultiSelect
      isAsync
      loadOptions={loadOptions}
      cacheOptions
      defaultOptions
      value={country}
      onChange={setCountry}
      placeholder="Select a country..."
      renderCustomSpinner={<Spinner size="md" />}
    />
  );
}
