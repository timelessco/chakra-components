### Basic Searchable Select

```jsx
function ComboBoxExample() {
  const fruits = [
    "Acai",
    "Apples",
    "Apricots",
    "Avocado",
    "Ackee",
    "Bananas",
    "Bilberries",
  ];

  const options = fruits.map((d, i) => ({
    value: d,
    label: d,
  }));

  return <MultiSelect options={options} />;
}
```
