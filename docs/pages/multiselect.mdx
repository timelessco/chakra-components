### Basic Select

```jsx
function ComboBoxExample() {
  const fruits = [
    "Acai",
    "Apples",
    "Apricots",
    "Avocado",
    "Ackee",
    "Bananas",
    "Bilberries",
  ];

  const options = fruits.map((d, i) => ({
    value: d,
    label: d,
  }));

  const handleFruitChange = fruit => {
    setFruit(fruit);
  };

  const [fruit, setFruit] = useState(null);

  return (
    <>
      <MultiSelect options={options} />
    </>
  );
}
```

    value: d,
    label: d,
  }));
